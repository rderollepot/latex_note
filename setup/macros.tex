% Boîte de résumé
\newtcolorbox{summarybox}[1]{
    colback=primaryColor!5!white,
    colframe=primaryColor,
    fonttitle=\bfseries,
    title=#1,
    sharp corners=south,
    boxrule=0.5mm
}

\newtcolorbox{infobox}[1]{
    colback=brandCyan!5!white,
    colframe=brandCyan,
    fonttitle=\bfseries,
    title=#1,
    sharp corners=south,
    boxrule=0.5mm
}

    colframe=red!75!black,
    fonttitle=\bfseries,
    title=#1,
    sharp corners=south,
    boxrule=0.5mm
}
